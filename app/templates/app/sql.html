WITH RECURSIVE recursive_cte AS (
    -- Anchor member (initial query)
    SELECT id, parent_id, name, 1 AS level
    FROM employee
    WHERE parent_id IS NULL  -- Root nodes condition
    
    UNION ALL
    
    -- Recursive member (subsequent queries)
    SELECT t.id, t.parent_id, t.name, rc.level + 1
    FROM employee t
    JOIN recursive_cte rc ON t.parent_id = rc.id
)
SELECT id, parent_id, name, level
FROM recursive_cte;
